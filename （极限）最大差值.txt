输入一串数字A1 A2 A3......An        
找出两个数 i,j（1≤i < j≤9n）使得 Aj - Ai 最大 //注意i j 大小关系
 

输入 #1复制
10
1
3
4
6
7
9
10
1
2
9

输出 #1复制
9




//我是真服了为什么输入里有特别小的负数
#define _CRT_SECURE_NO_WARNINGS
#include<cstdio>
#include<algorithm>
using namespace std;
int a[1000005];
inline int read() {
	int s = 0, w = 1;
	char ch = getchar();
	while (ch < '0' || ch>'9') { if (ch == '-') w = -1; ch = getchar(); }
	while (ch >= '0' && ch <= '9') s = s * 10 + ch - '0', ch = getchar();
	return s * w;
}
int main()
{
	int n;
	n = read();
	long long maxx = -1e10, c , t;
	c = read(); n--;
	while (n--) {
		t = read();
		if (t - c > maxx) {
			maxx = t - c;
		}
		if (t < c) c = t;  //最小值不断更新
	}
	printf("%lld", maxx);
}
